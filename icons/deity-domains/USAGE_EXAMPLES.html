Example 6: JavaScript Integration

How to use the DeityDomainIcons API:

// Initialize
await window.deityDomainIcons.init();

// Get icon for a domain
const warIcon = await window.deityDomainIcons.getIconSVG('war');

// Get icon for deity based on domains
const deity = {
    name: "Zeus",
    domains: ["sky", "thunder", "justice"]
};
const icon = await window.deityDomainIcons.getDeityIcon(deity);

// Render domain badges with icons
const html = await window.deityDomainIcons.renderDomainIcons(['war', 'wisdom']);
container.innerHTML = html;
