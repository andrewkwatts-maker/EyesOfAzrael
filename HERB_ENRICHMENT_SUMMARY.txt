THE 7 METADATA SECTIONS
================================================================================

1. PROPERTIES
   Structure:
   - magical: string[]       (Spiritual/mystical properties)
   - medicinal: string[]     (Health benefits)
   - spiritual: string[]     (Ritual/practice aspects)

   Example:
   "properties": {
     "magical": ["purity", "enlightenment", "rebirth"],
     "medicinal": ["adaptogenic", "sedative", "nutritive"],
     "spiritual": ["meditation anchor", "dharma practice"]
   }

2. PREPARATIONS
   Structure:
   - primary: string[]       (Main methods with instructions)
   - alternative: string[]   (Variant methods)
   - dosage: string         (Recommended amounts/frequency)

   Example:
   "preparations": {
     "primary": ["Lotus seed tea: simmer 10 minutes"],
     "alternative": ["Fresh root: slice and cook"],
     "dosage": "6-15g dried seeds daily"
   }

3. ASSOCIATIONS
   Structure:
   - deities: string[]       (Deity entity IDs)
   - concepts: string[]      (Universal ideas)
   - elements: string[]      (fire, water, earth, air, space)
   - chakras: string[]       (root through crown)

   Example:
   "associations": {
     "deities": ["gautama_buddha", "avalokiteshvara"],
     "concepts": ["enlightenment", "compassion"],
     "elements": ["water", "earth"],
     "chakras": ["heart", "crown"]
   }

4. HARVESTING
   Structure:
   - season: string          (When to harvest)
   - method: string          (How to harvest)
   - conditions: string      (Growing requirements)

   Example:
   "harvesting": {
     "season": "Summer (July-September for flowers)",
     "method": "Hand-harvest fully opened flowers",
     "conditions": "Shallow ponds, full sun, 6-18 inches water depth"
   }

5. DANGERS
   Structure:
   - toxicity: string        (Non-toxic, Low, Moderate, High)
   - warnings: string[]      (Safety precautions)
   - contraindications: string[] (Who should avoid)

   Example:
   "dangers": {
     "toxicity": "Non-toxic",
     "warnings": ["Ensure clean water if wild-harvesting"],
     "contraindications": ["Generally safe as food and tea"]
   }

6. SUBSTITUTES
   Structure: Array of objects
   - name: string            (Alternative herb name)
   - reason: string          (Why it works as substitute)
   - tradition: string       (Cultural context)

   Example:
   "substitutes": [
     {
       "name": "Water Lily",
       "reason": "Similar aquatic flower, shared spiritual properties",
       "tradition": "multiple"
     }
   ]

7. BOTANICAL INFO
   Structure:
   - scientificName: string  (Latin binomial)
   - family: string          (Plant family)
   - nativeRegion: string    (Geographic origin)
   - commonNames: string[]   (Alternative names)

   Example:
   "botanicalInfo": {
     "scientificName": "Nelumbo nucifera",
     "family": "Nelumbonaceae",
     "nativeRegion": "Asia (India to Japan, Australia)",
     "commonNames": ["Sacred Lotus", "Indian Lotus", "Padma"]
   }
